---
title: 'Shipment'
---

<ResponseField name="id" type="integer" required>
    ID of the shipment.
</ResponseField>

<ResponseField name="status" type="integer" required>
    A [Status](/guide/enums/shipment-statuses) enum representing the status of the shipment.
</ResponseField>

<ResponseField name="package" type="string" required>
    The name of the package template used for the shipment.
</ResponseField>

<ResponseField name="length" type="integer" required>
    The length of the shipment (in inches, rounded up).
</ResponseField>

<ResponseField name="width" type="integer" required>
    The width of the shipment (in inches, rounded up).
</ResponseField>

<ResponseField name="height" type="integer" required>
    The height of the shipment (in inches, rounded up).
</ResponseField>

<ResponseField name="weight" type="integer" required>
    The weight of the shipment (in pounds, rounded up).
</ResponseField>

<ResponseField name="tracking_number" type="string">
    The tracking number for the shipment.
</ResponseField>

<ResponseField name="tracking_url" type="string">
    The URL to the tracking page for the shipment.
</ResponseField>

<ResponseField name="label_url" type="string">
    The URL to the shipping label for the shipment.
</ResponseField>

<ResponseField name="created_at" type="string" required>
    Timestamp of when the shipment was created.
</ResponseField>

<ResponseField name="updated_at" type="string" required>
    Timestamp of when the shipment was last updated.
</ResponseField>
